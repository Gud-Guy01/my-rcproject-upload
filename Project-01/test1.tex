% ============================================
\makeatletter

% Create storage for marks
\newtoks\LT@savedmark
\LT@savedmark=\expandafter{\leftmark}

% Patch section command to save mark
\pretocmd{\section}{%
    \LT@savedmark=\expandafter{\leftmark}%
}{}{}

% Patch longtable's start
\pretocmd{\LT@start}{%
    \LT@savedmark=\expandafter{\leftmark}%
}{}{}

% Replace every occurrence of \markright with \markboth in output
\patchcmd{\LT@output}{%
    \copy\LT@head\nobreak
}{%
    \copy\LT@head\nobreak
    \markboth{\the\LT@savedmark}{\the\LT@savedmark}%
}{}{}

% Also patch the final page case
\patchcmd{\LT@output}{%
    \vbox
        {\unvbox\z@\box\ifvoid\LT@lastfoot\LT@foot\else\LT@lastfoot\fi}%
}{%
    \vbox
        {\unvbox\z@\box\ifvoid\LT@lastfoot\LT@foot\else\LT@lastfoot\fi}%
    \markboth{\the\LT@savedmark}{\the\LT@savedmark}%
}{}{}

% Force mark at every table header
\pretocmd{\endhead}{%
    \noalign{\markboth{\the\LT@savedmark}{\the\LT@savedmark}}%
}{}{}

\makeatother
